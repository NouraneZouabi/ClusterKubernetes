
# This documentation is issued from Docker official documentation at "https://docs.docker.com/engine/install/ubuntu/".
# In this section, we are going to install containerd on ubuntu as a container runtime interface(you can work with cri-o or any other CRI):

===> VERY IMPORTANT: kubernetes does not support docker container runtime because docker runtime does not implement the CRI interface.

=> Uninstall docker (if exists):
sudo apt-get remove docker docker-engine docker.io containerd runc

=> Update apt packages:
sudo apt-get update
sudo apt-get install ca-certificates curl gnupg lsb-release

=> Add Dockerâ€™s official GPG key:
sudo mkdir -m 0755 -p /etc/apt/keyrings
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /etc/apt/keyrings/docker.gpg

=> Set up the repository:
echo "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

=> Update the apt package index:
sudo apt-get update
-> if you got an error during the update, use:
sudo chmod a+r /etc/apt/keyrings/docker.gpg
sudo apt-get update

=> install the containerd package:
sudo apt-get install containerd.io

=> Verify that containerd service is running:
systemctl status containerd





# This documentation is issued from kubernetes official documentation at "https://kubernetes.io/docs/setup/production-environment/container-runtimes/"
# After successfully installing CRI in our VMs, now we are going to configure network dependencies in all of the VMs.

===> Install Required Packages
=> Forwarding IPv4 and letting iptables see bridged traffic:
cat <<EOF | sudo tee /etc/modules-load.d/k8s.conf
overlay
br_netfilter
EOF

sudo modprobe overlay
sudo modprobe br_netfilter

=> Sysctl params required by setup, params persist across reboots:
cat <<EOF | sudo tee /etc/sysctl.d/k8s.conf
net.bridge.bridge-nf-call-iptables  = 1
net.bridge.bridge-nf-call-ip6tables = 1
net.ipv4.ip_forward = 1
EOF

=> Apply sysctl params without reboot:
sudo sysctl --system

=> Verify that the br_netfilter, overlay modules are loaded by running below instructions:
lsmod | grep br_netfilter
lsmod | grep overlay

=> Verify system variables(net.bridge.bridge-nf-call-iptables, net.bridge.bridge-nf-call-ip6tables, net.ipv4.ip_forward) are set to 1 :
sysctl net.bridge.bridge-nf-call-iptables net.bridge.bridge-nf-call-ip6tables net.ipv4.ip_forward


===> VERY IMPORTANT: We should configure "systemd" as a cgroup for containerd.

=> Open containerd configuration file located at: /etc/containerd/config.toml
sudo vim /etc/containerd/config.toml

=> Delete the entire content of the file and replace it with :
#=============================================================================== Don't copy this line.
[plugins."io.containerd.grpc.v1.cri".containerd.runtimes.runc]
  [plugins."io.containerd.grpc.v1.cri".containerd.runtimes.runc.options]
    SystemdCgroup = true
#===============================================================================  Don't copy this line.

=> Restart the containerd service:
sudo systemctl restart containerd

=> Check the status of containerd service:
systemctl status containerd
